```typescript
import { Router } from 'express';
import type { Request, Response, NextFunction } from 'express';
import { knex } from '../db';

const router = Router();

// GET /api/entities
router.get('/', async (_req: Request, res: Response, next: NextFunction) => {
  try {
    const entities = await knex('entities').select('*').orderBy('id');
    res.json(entities);
  } catch (err) { next(err); }
});

// GET /api/entities/:id
router.get('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const entity = await knex('entities').where({ id: req.params.id }).first();
    if (!entity) { const e: any = new Error('Entity not found'); e.status = 404; throw e; }
    res.json(entity);
  } catch (err) { next(err); }
});

// POST /api/entities
router.post('/', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { type, name, description } = req.body;
    if (!type || !name) { const e: any = new Error('type and name are required'); e.status = 400; throw e; }
    const [created] = await knex('entities').insert({ type, name, description }).returning('*');
    res.status(201).json(created);
  } catch (err) { next(err); }
});

// PUT /api/entities/:id
router.put('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { type, name, description } = req.body;
    const [updated] = await knex('entities').where({ id: req.params.id }).update({ type, name, description, updated_at: knex.fn.now() }).returning('*');
    if (!updated) { const e: any = new Error('Entity not found'); e.status = 404; throw e; }
    res.json(updated);
  } catch (err) { next(err); }
});

// DELETE /api/entities/:id
router.delete('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const deleted = await knex('entities').where({ id: req.params.id }).delete();
    if (!deleted) { const e: any = new Error('Entity not found'); e.status = 404; throw e; }
    res.status(204).send();
  } catch (err) { next(err); }
});

export default router;
```
