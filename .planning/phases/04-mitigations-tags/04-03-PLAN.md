import { Router, Request, Response, NextFunction } from 'express';
import { db } from '../db';

export const tagsRouter = Router();

// GET /api/tags — list all tags
tagsRouter.get('/', async (_req: Request, res: Response, next: NextFunction) => {
  try {
    const tags = await db('tags').select('*').orderBy('name', 'asc');
    res.json(tags);
  } catch (err) {
    next(err);
  }
});

// GET /api/tags/overview — per-tag capital exposure
// IMPORTANT: this route must be declared BEFORE /:id
tagsRouter.get('/overview', async (_req: Request, res: Response, next: NextFunction) => {
  try {
    // Total portfolio value for percentage calculation
    const [{ total }] = await db('assets').sum({ total: 'value' });
    const totalValue = parseFloat(total as string) || 0;

    const tags = await db('tags').select('id', 'name').orderBy('name', 'asc');

    const result = await Promise.all(
      tags.map(async (tag) => {
        const assets = await db('assets as a')
          .join('asset_tags as at', 'a.id', 'at.asset_id')
          .where('at.tag_id', tag.id)
          .select('a.id', 'a.name', 'a.value', 'a.currency')
          .orderBy('a.name', 'asc');

        const tagTotal = assets.reduce((sum, a) => sum + parseFloat(a.value as string), 0);
        const portfolioPct = totalValue > 0 ? Math.round((tagTotal / totalValue) * 10000) / 100 : 0;

        return {
          tag_id: tag.id,
          tag_name: tag.name,
          asset_count: assets.length,
          total_value: tagTotal,
          portfolio_pct: portfolioPct,
          assets: assets.map((a) => ({
            id: a.id,
            name: a.name,
            value: parseFloat(a.value as string),
            currency: a.currency,
            capital_weight_pct: totalValue > 0 ? Math.round((parseFloat(a.value as string) / totalValue) * 10000) / 100 : 0,
          })),
        };
      }),
    );

    res.json(result);
  } catch (err) {
    next(err);
  }
});

// POST /api/tags — create tag
tagsRouter.post('/', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { name } = req.body as { name: string };
    if (!name || name.trim() === '') {
      res.status(400).json({ error: { message: 'name is required', status: 400 } });
      return;
    }

    const [created] = await db('tags')
      .insert({ name: name.trim() })
      .returning('*');

    res.status(201).json(created);
  } catch (err: unknown) {
    // Unique constraint violation
    if ((err as { code?: string }).code === '23505') {
      res.status(409).json({ error: { message: 'A tag with that name already exists', status: 409 } });
      return;
    }
    next(err);
  }
});

// DELETE /api/tags/:id — delete tag (cascades asset_tags)
tagsRouter.delete('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const deleted = await db('tags').where({ id: req.params.id }).delete();
    if (!deleted) {
      res.status(404).json({ error: { message: 'Tag not found', status: 404 } });
      return;
    }
    res.status(204).send();
  } catch (err) {
    next(err);
  }
});

import { Router, Request, Response, NextFunction } from 'express';
import { db } from '../db';

// mergeParams: true so :id from assetsRouter is accessible
export const assetTagsRouter = Router({ mergeParams: true });

// GET /api/assets/:id/tags — list tags assigned to asset
assetTagsRouter.get('/', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const assetId = req.params.id;

    const asset = await db('assets').where({ id: assetId }).first();
    if (!asset) {
      res.status(404).json({ error: { message: 'Asset not found', status: 404 } });
      return;
    }

    const tags = await db('tags as t')
      .join('asset_tags as at', 't.id', 'at.tag_id')
      .where('at.asset_id', assetId)
      .select('t.id as tag_id', 't.name as tag_name')
      .orderBy('t.name', 'asc');

    res.json(tags);
  } catch (err) {
    next(err);
  }
});

// POST /api/assets/:id/tags — assign tag to asset
assetTagsRouter.post('/', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const assetId = req.params.id;
    const { tag_id } = req.body as { tag_id: string };

    if (!tag_id) {
      res.status(400).json({ error: { message: 'tag_id is required', status: 400 } });
      return;
    }

    const [asset, tag] = await Promise.all([
      db('assets').where({ id: assetId }).first(),
      db('tags').where({ id: tag_id }).first(),
    ]);

    if (!asset) {
      res.status(404).json({ error: { message: 'Asset not found', status: 404 } });
      return;
    }
    if (!tag) {
      res.status(404).json({ error: { message: 'Tag not found', status: 404 } });
      return;
    }

    // Upsert — idempotent assignment
    await db('asset_tags')
      .insert({ asset_id: assetId, tag_id })
      .onConflict(['asset_id', 'tag_id'])
      .ignore();

    res.status(201).json({ tag_id, tag_name: tag.name });
  } catch (err) {
    next(err);
  }
});

// DELETE /api/assets/:id/tags/:tagId — remove tag from asset
assetTagsRouter.delete('/:tagId', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { id: assetId, tagId } = req.params;

    const deleted = await db('asset_tags')
      .where({ asset_id: assetId, tag_id: tagId })
      .delete();

    if (!deleted) {
      res.status(404).json({ error: { message: 'Tag not assigned to asset', status: 404 } });
      return;
    }

    res.status(204).send();
  } catch (err) {
    next(err);
  }
});
