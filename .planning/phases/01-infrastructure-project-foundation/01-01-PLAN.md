<task type="auto">
  <name>Task 1: Create root docker-compose.yml</name>
  <files>docker-compose.yml</files>
  <action>Create docker-compose.yml at the project root with the following exact structure:

```yaml
version: "3.9"

services:
  postgres:
    image: postgres:16-alpine
    container_name: precious_postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: ${POSTGRES_DB:-precious}
      POSTGRES_USER: ${POSTGRES_USER:-precious}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-precious}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    networks:
      - precious_net
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-precious} -d ${POSTGRES_DB:-precious}"]
      interval: 5s
      timeout: 5s
      retries: 5

  api:
    build:
      context: ./api
      dockerfile: Dockerfile
    container_name: precious_api
    restart: unless-stopped
    environment:
      NODE_ENV: production
      PORT: ${API_PORT:-3001}
      DATABASE_URL: postgresql://${POSTGRES_USER:-precious}:${POSTGRES_PASSWORD:-precious}@postgres:5432/${POSTGRES_DB:-precious}
    ports:
      - "${API_PORT:-3001}:3001"
    depends_on:
      postgres:
        condition: service_healthy
    networks:
      - precious_net

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: precious_frontend
    restart: unless-stopped
    environment:
      VITE_API_URL: ${VITE_API_URL:-http://localhost:3001}
    ports:
      - "${FRONTEND_PORT:-3000}:80"
    depends_on:
      - api
    networks:
      - precious_net

volumes:
  postgres_data:
    name: precious_postgres_data

networks:
  precious_net:
    name: precious_net
    driver: bridge
```
  </action>
  <verify>grep -q "postgres_data" docker-compose.yml && grep -q "precious_net" docker-compose.yml && grep -q "service_healthy" docker-compose.yml && echo "OK"</verify>
  <acceptance_criteria>
    - "docker-compose.yml contains 3 services: postgres, api, frontend"
    - "docker-compose.yml contains named volume `postgres_data`"
    - "docker-compose.yml contains network `precious_net`"
    - "api depends_on postgres with condition: service_healthy"
  </acceptance_criteria>
  <done>docker-compose.yml exists with all 3 services, named volume, shared network, and health-checked dependency chain</done>
</task>

```yaml
  api:
    build:
      context: ./api
      dockerfile: Dockerfile.dev
    environment:
      NODE_ENV: development

      - ./api:/app
      - /app/node_modules
    command: npm run dev

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile.dev
    environment:
      VITE_API_URL: http://localhost:3001
    volumes:
      - ./frontend:/app
      - /app/node_modules

      - "5173:5173"
```
