Create `docker-compose.yml`. Key requirements:
- db service uses healthcheck so api waits for PostgreSQL to be ready (prevents migration race condition)
- API env uses Docker service name "db" in DATABASE_URL
- Frontend env sets VITE_API_URL to http://api:3001 (Docker service name)
- Host volumes mount src/ directories for hot-reload without npm ci per change
- data/db volume mount path: ./data/db (already exists in repo)

```yaml
services:
  db:
    image: postgres:16
    restart: unless-stopped
    environment:
      POSTGRES_DB: precious
      POSTGRES_USER: precious
      POSTGRES_PASSWORD: precious
    volumes:
      - ./data/db:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U precious -d precious"]
      interval: 5s
      timeout: 5s
      retries: 10

  api:
    build:
      context: ./api
      dockerfile: Dockerfile
    restart: unless-stopped
    ports:
      - "3001:3001"
    environment:
      DATABASE_URL: postgres://precious:precious@db:5432/precious
      NODE_ENV: development
      PORT: 3001
    volumes:
      - ./api/src:/app/src
      - ./api/migrations:/app/migrations
    depends_on:
      db:
        condition: service_healthy

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      VITE_API_URL: http://api:3001
    volumes:
      - ./frontend/src:/app/src
      - ./frontend/public:/app/public
      - ./frontend/index.html:/app/index.html
    depends_on:
      - api
```

Create `Makefile`. Use tabs (not spaces) for recipe indentation — Make requires literal tab characters.

```makefile
.PHONY: dev up down clean logs-api migrate

# Start full stack with hot reload (frontend :3000, API :3001)
dev:
	docker compose up --build

# Start production stack detached
up:
	docker compose up --build -d

# Stop containers
down:
	docker compose down

# Stop containers + remove volumes (full reset)
clean:
	docker compose down -v

# Tail API logs
logs-api:
	docker compose logs -f api

# Run migrations manually (normally automatic on API startup)
migrate:
	docker compose exec api npm run migrate
```

IMPORTANT: The indentation before each `docker compose` command MUST be a literal tab character, not spaces. If the file uses spaces, `make` will error with "missing separator".

```typescript
/**
 * Central API client.
 * ALL fetch calls in the frontend MUST use api.* methods — never inline fetch().
 * Error shape from API: { error: { message: string, status: number } }
 */

async function request<T>(path: string, options?: RequestInit): Promise<T> {
  const res = await fetch(`/api${path}`, {
    headers: {
      'Content-Type': 'application/json',
      ...options?.headers,
    },
    ...options,
  });

  if (!res.ok) {
    const body = await res
      .json()
      .catch(() => ({ error: { message: 'Network error', status: res.status } }));
    throw new Error(body.error?.message ?? 'Request failed');
  }

  return res.json() as Promise<T>;
}

// Phase 2 will expand this with entity namespaces:
// api.assets.list(), api.assets.create(), etc.
export const api = {
  health: {
    // /health is NOT under /api — fetch directly (not via request())
    get: (): Promise<{ status: string }> =>
      fetch('/health').then((r) => r.json()),
  },
};

export { request };
```
