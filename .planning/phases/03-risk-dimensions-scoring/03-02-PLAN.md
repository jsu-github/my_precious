```typescript
import { Router, Request, Response, NextFunction } from 'express';
import { db } from '../db';

export const dimensionsRouter = Router();

// GET /api/dimensions — list all, defaults first then alphabetical
dimensionsRouter.get('/', async (_req: Request, res: Response, next: NextFunction) => {
  try {
    const dimensions = await db('risk_dimensions')
      .select('*')
      .orderBy([{ column: 'is_default', order: 'desc' }, { column: 'name', order: 'asc' }]);
    res.json(dimensions);
  } catch (err) {
    next(err);
  }
});

// POST /api/dimensions — create custom dimension
dimensionsRouter.post('/', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { name, description } = req.body as { name: string; description?: string };

    if (!name || name.trim() === '') {
      res.status(400).json({ error: { message: 'name is required', status: 400 } });
      return;
    }

    const [created] = await db('risk_dimensions')
      .insert({
        name: name.trim(),
        description: description?.trim() ?? null,
        is_default: false,
      })
      .returning('*');

    res.status(201).json(created);
  } catch (err) {
    next(err);
  }
});

// PUT /api/dimensions/:id — update name/description (all dimensions, including defaults)
dimensionsRouter.put('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const { name, description } = req.body as { name: string; description?: string };

    if (!name || name.trim() === '') {
      res.status(400).json({ error: { message: 'name is required', status: 400 } });
      return;
    }

    const [updated] = await db('risk_dimensions')
      .where({ id: req.params.id })
      .update({
        name: name.trim(),
        description: description?.trim() ?? null,
        updated_at: db.fn.now(),
      })
      .returning('*');

    if (!updated) {
      res.status(404).json({ error: { message: 'Dimension not found', status: 404 } });
      return;
    }

    res.json(updated);
  } catch (err) {
    next(err);
  }
});

// DELETE /api/dimensions/:id — reject if is_default=true
dimensionsRouter.delete('/:id', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const dimension = await db('risk_dimensions').where({ id: req.params.id }).first();

    if (!dimension) {
      res.status(404).json({ error: { message: 'Dimension not found', status: 404 } });
      return;
    }

    if (dimension.is_default) {
      res.status(400).json({ error: { message: 'Default dimensions cannot be deleted', status: 400 } });
      return;
    }

    await db('risk_dimensions').where({ id: req.params.id }).delete();
    res.status(204).send();
  } catch (err) {
    next(err);
  }
});
```
