// GET /api/assets — all assets with entity + location joined
router.get('/', async (_req, res, next) => {
  try {
    const assets = await knex('assets')
      .leftJoin('entities', 'assets.entity_id', 'entities.id')
      .leftJoin('asset_locations', 'assets.location_id', 'asset_locations.id')
      .select(
        'assets.*',
        'entities.name as entity_name',
        'entities.type as entity_type',
        'asset_locations.name as location_name',
        'asset_locations.custodian_name',
      )
      .orderBy('assets.id');
    res.json(assets);
  } catch (err) { next(err); }
});

// GET /api/assets?entity_id=N — filter by entity
// Add to the GET / handler: if (req.query.entity_id) query.where('assets.entity_id', req.query.entity_id)

// GET /api/assets/:id — single asset with acquisitions
router.get('/:id', async (req, res, next) => {
  try {
    const asset = await knex('assets')
      .leftJoin('entities', 'assets.entity_id', 'entities.id')
      .leftJoin('asset_locations', 'assets.location_id', 'asset_locations.id')
      .select('assets.*', 'entities.name as entity_name', 'asset_locations.name as location_name', 'asset_locations.custodian_name')
      .where('assets.id', req.params.id)
      .first();
    if (!asset) { const e: any = new Error('Asset not found'); e.status = 404; throw e; }
    const acquisitions = await knex('acquisitions').where({ asset_id: req.params.id }).orderBy('purchase_date', 'desc');
    res.json({ ...asset, acquisitions });
  } catch (err) { next(err); }
});
