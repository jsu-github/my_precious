Add `debounceRef` for the threshold input: `const debounceRef = React.useRef<ReturnType<typeof setTimeout> | null>(null);`

**3. Extend `fetchData`** to also fetch warnings and threshold alongside the summary:
```ts
const fetchData = useCallback(() => {
  setRefreshing(true);
  Promise.all([
    api.portfolio.summary(),
    api.portfolio.warnings(),
    api.portfolio.getThreshold(),
  ])
    .then(([summary, warns, thresh]) => {
      setData(summary);
      setWarnings(warns);
      setThreshold(thresh.threshold);
      setThresholdInput(String(thresh.threshold));
    })
    .catch(() => setError('Failed to load portfolio summary'))
    .finally(() => { setLoading(false); setRefreshing(false); });
}, []);
```

**4. Add threshold change handler**:
```ts
const handleThresholdChange = (value: string) => {
  setThresholdInput(value);
  if (debounceRef.current) clearTimeout(debounceRef.current);
  debounceRef.current = setTimeout(() => {
    const n = parseInt(value, 10);
    if (Number.isInteger(n) && n >= 1 && n <= 99) {
      setThreshold(n);
      api.portfolio.warnings().then(setWarnings).catch(() => null);
    }
  }, 300);
};

const handleThresholdBlur = () => {
  const n = parseInt(thresholdInput, 10);
  const clamped = Math.max(1, Math.min(99, isNaN(n) ? 70 : n));
  setThresholdInput(String(clamped));
  setThreshold(clamped);
  api.portfolio.setThreshold(clamped).catch(() => null);
  api.portfolio.warnings().then(setWarnings).catch(() => null);
};
```
